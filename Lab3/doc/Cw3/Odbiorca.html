def ilosc_liczb_parzystych()

    array.keep_if { |number| number.to_f % 2 == 0 }
    return array.length
end

def utworz_tablice(n,m)
    array = Array.new(n)
    i = 0
    while i < n
        array[i] = rand(m - 1) + 1
        i += 1
    end
    return array
end
